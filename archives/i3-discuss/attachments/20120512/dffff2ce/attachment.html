.py:<br>
#!/usr/bin/env python<br>
# -*- coding: utf-8 -*-<br>
<br>
import optparse<br>
<br>
BR_FILE = '/sys/class/backlight/acpi_video0/brightness'<br>
BR_MAX = 10<br>
BR_MIN = 0<br>
<br>
<br>
def br_set(value):<br>
    open(BR_FILE,"w").write("%d\n" % value)<br>
<br>
<br>
def br_increase(current):<br>
    if current >= BR_MAX:<br>
        print "can not increase more then %d" % BR_MAX<br>
        return<br>
    br_set((current+1))<br>
<br>
def br_decrease(current):<br>
    if current <= BR_MIN:<br>
        print "can not decrease less then %d" % BR_MIN<br>
        return<br>
    br_set((current-1))<br>
<br>
<br>
def main():<br>
    p = optparse.OptionParser()<br>
    p.add_option("--increase", '-i', action='store_true')<br>
    p.add_option("--decrease", '-d', action='store_true')<br>
    p.add_option("--set", '-s')<br>
    options, arguments = p.parse_args()<br>
    current = int(open(BR_FILE).read())<br>
    if options.increase:<br>
        br_increase(current)<br>
    elif options.decrease:<br>
        br_decrease(current)<br>
    elif options.set:<br>
        br_set(int(options.set))<br>
<br>
<br>
if __name__ == '__main__':<br>
    main()<br>
