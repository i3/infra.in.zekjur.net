Just a note for those who want to do the same, here's my ~/.conkyrc

background no
use_xft yes
xftfont Terminus:size=9
update_interval 1.0
total_run_times 0
# the next two lines are very important, without them i3 won't give space for conky
own_window yes
own_window_type dock
own_window_transparent yes
double_buffer yes
# I use the next two lines to make the conky line as thin as possible to waste as little space as possible
border_inner_margin 0
border_outer_margin 0
