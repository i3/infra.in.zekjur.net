Thanks! Works fine, here is script which worked for me:

#!/bin/sh
# shell script to prepend i3status with more stuff

i3status -c ~/.i3/i3status.conf | while :
do
    read line
    mystuff=`~/bin/measure-net-speed.sh`
    echo "$mystuff | $line"
done
